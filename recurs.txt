#1 Дано натуральное число. Вычислите сумму его цифр.

def nat(n):
    """
    Функция принимает число и возращает сумму его чисел
    """
    if n == 0:
        return 0
    else:
        return n % 10 + nat(n // 10)







#2 Дано натуральное число. Записать число в обратном порядке

#3	Проверить число на простоту
def prosto(n, d = 2):
    """
    функция принмает число и
    проверяет его на простоту
    """
    if n < 2:
        return False
    if d * d > n:
        return True
    if n % d == 0:
        return False
    return prosto(n, d + 1)




#4.	Написать функцию, которая будет принимать натуральное число n и
# возвращать n-ую строку треугольника Паскаля.

def pascal(n):
    """
    функция принимает число и возвращаетстроку
    и треугольника паскаля ровный этому числу
    """
    if n == 0:
        return [1]
    ps = pascal(n - 1)
    pas = [1]
    for i in range(len(ps) - 1):
        pas += [ps[i] + ps[i + 1]]
    pas += [1]
    return pas
