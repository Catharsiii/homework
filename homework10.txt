def proverka(v1, v2):
    '''
    Функция принимает в качестве аргументов два списка из чисел
    и проверяет,являются ли частные соответсвенных координат равными
    '''

    flag = 0
    tup = list(map(lambda v1, v2: (v1, v2), v1, v2))
    delimoe = list(map(lambda tup: tup[0], tup))
    delitel = list(map(lambda tup: tup[1], tup))
    #print(tup, '\n', delimoe, '\n', delitel)
    coeff = []
    for i in range(len(tup)):
        coeff.append(delimoe[i] / delitel[i])
    if coeff[0] == coeff[1] and coeff[2] == coeff[-1] and coeff[1] == coeff[2]:
        flag = 1 
    else:
        return "T"
